111 
112  erhic::LeptonKinematicsComputer mcDis(*mcEvent);

115  unsigned mcIx = mg->GetXaxis()->FindBin(mcKin->mX) - 1;

123  if (rcIx >=0 && rcIx < _X_BNUM_ &&
124  rcIq >=0 && rcIq < _Q2_BNUM_)
