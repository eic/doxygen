98 // ----- Accessor to covariance matrix --------------------------------
99 void CbmVertex::CovMatrix(TMatrixFSym& covMat) const {
100  Int_t index = 0;
101  for (int i=0; i<3; i++) {
102  for (int j=i; j<3; j++) {
103  covMat[i][j] = fCovMatrix[index];
104  covMat[j][i] = fCovMatrix[index];
105  index++;
106  }
107  }
108 }
