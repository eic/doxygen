310 void PHActsTrkProp::updateSvtxTrack(Trajectory traj,
311  const unsigned int trackKey,
312  Acts::Vector3D vertex)
313 {
314 
315  ActsTransformations *rotater = new ActsTransformations();
316  rotater->setVerbosity(0);
317 
318  int iTrack = 0;
319 
323  SvtxTrack *origTrack = m_trackMap->find(trackKey)->second;
324 
327  unsigned int vertexId = origTrack->get_vertex_id();
328 
330  if(vertexId == UINT_MAX)
331  vertexId = 0;
332 
333  origTrack->Reset();
334 
335 
337  float pathLength = 0.0;
338  SvtxTrackState_v1 out(pathLength);
339  out.set_x(0.0);
340  out.set_y(0.0);
341  out.set_z(0.0);
342  origTrack->insert_state(&out);
343 
346  const auto& [trackTips, mj] = traj.trajectory();
347 
348  if(trackTips.empty())
349  {
350  if(Verbosity() > 5)
351  std::cout << "Empty multiTrajectory..." << std::endl;
352  return;
353  }
354 
355  std::map<const size_t, const unsigned int> trackTipTrackKeyMap;
356 
358  for(const size_t& trackTip : trackTips)
359  {
360  auto trajState = Acts::MultiTrajectoryHelpers::trajectoryState(mj, trackTip);
361 
362  size_t nStates = trajState.nStates;
363  size_t nMeasurements = trajState.nMeasurements;
364  size_t nOutliers = trajState.nOutliers;
365  size_t nHoles = trajState.nHoles;
366  double chi2sum = trajState.chi2Sum;
367  size_t NDF = trajState.NDF;
368 
370  if( !traj.hasTrackParameters(trackTip) )
371  continue;
372 
373  const auto& fittedParameters = traj.trackParameters(trackTip);
374 
375  float x = fittedParameters.position(m_tGeometry->geoContext)(0)
376  / Acts::UnitConstants::cm;
377  float y = fittedParameters.position(m_tGeometry->geoContext)(1)
378  / Acts::UnitConstants::cm;
379  float z = fittedParameters.position(m_tGeometry->geoContext)(2)
380  / Acts::UnitConstants::cm;
381  float px = fittedParameters.momentum()(0);
382  float py = fittedParameters.momentum()(1);
383  float pz = fittedParameters.momentum()(2);
384 
385  if(Verbosity() > 0)
386  {
387  std::cout << "Track fit returned a track with: " << std::endl
388  << "momentum : "
389  << fittedParameters.momentum().transpose()<< std::endl
390  << "position : "
391  << fittedParameters.position(m_tGeometry->geoContext).transpose()
392  << std::endl
393  << "charge : " << fittedParameters.charge() << std::endl;
394  std::cout << "Track has " << nStates << " states and "
395  << nMeasurements << " measurements and "
396  << nHoles << " holes and " << nOutliers << " outliers "
397  << std::endl;
398  }
399 
400  float qOp = fittedParameters.parameters()[Acts::eBoundQOverP];
401 
402  Acts::BoundSymMatrix rotatedCov = Acts::BoundSymMatrix::Zero();
403  float DCA3Dxy = NAN;
404  float DCA3Dz = NAN;
405  float DCA3DxyCov = NAN;
406  float DCA3DzCov = NAN;
407 
408  if(fittedParameters.covariance())
409  {
410  rotater->setVerbosity(0);
411  rotatedCov = rotater->rotateActsCovToSvtxTrack(fittedParameters,
412  m_tGeometry->geoContext);
413 
414 
415 
416  rotater->calculateDCA(fittedParameters, vertex,
417  rotatedCov,
418  m_tGeometry->geoContext,
419  DCA3Dxy, DCA3Dz, DCA3DxyCov, DCA3DzCov);
420  }
422  if(iTrack == 0)
423  {
424  if(Verbosity() > 0)
425  std::cout << "Replacing SvtxTrack " << trackKey
426  << " with parameters from trackTip " << trackTip
427  << std::endl;
428 
429  origTrack->set_id(trackKey);
430  origTrack->set_vertex_id(vertexId);
431  origTrack->set_chisq(chi2sum);
432  origTrack->set_ndf(NDF);
433  origTrack->set_charge(1);
434 
435  if(qOp < 0)
436  origTrack->set_charge(-1);
437  for(int i = 0; i < 6; ++i)
438  {
439  for(int j = 0; j < 6; ++j)
440  {
441  origTrack->set_error(i,j, rotatedCov(i,j));
442  }
443  }
444 
445  origTrack->set_px(px);
446  origTrack->set_py(py);
447  origTrack->set_pz(pz);
448  origTrack->set_x(x);
449  origTrack->set_y(y);
450  origTrack->set_z(z);
451  origTrack->set_dca3d_xy(DCA3Dxy / Acts::UnitConstants::cm);
452  origTrack->set_dca3d_z(DCA3Dz / Acts::UnitConstants::cm);
453  origTrack->set_dca3d_xy_error(DCA3DxyCov / Acts::UnitConstants::cm);
454  origTrack->set_dca3d_z_error(DCA3DzCov / Acts::UnitConstants::cm);
455  rotater->setVerbosity(0);
456  rotater->fillSvtxTrackStates(traj, trackTip, origTrack,
457  m_tGeometry->geoContext,
458  m_hitIdClusKey);
459 
460  trackTipTrackKeyMap.insert(std::pair<const size_t, const unsigned int>
461  (trackTip, trackKey));
462 
463 
464  }
465  else
466  {
468 
469  const unsigned int lastTrackKey = m_trackMap->end()->first;
470 
472  if(Verbosity() > 0)
473  std::cout << "Creating new SvtxTrack with trackKey " << lastTrackKey
474  << " corresponding to trackTip " << trackTip << std::endl;
475 
476 
477  SvtxTrack_v1 *newTrack = new SvtxTrack_v1();
479  newTrack->set_id(lastTrackKey);
480  newTrack->set_vertex_id(vertexId);
481  newTrack->set_chisq(chi2sum);
482  newTrack->set_ndf(NDF);
483  newTrack->set_charge(1);
485  if(qOp < 0)
486  newTrack->set_charge(-1);
487 
488  newTrack->set_px(px);
489  newTrack->set_py(py);
490  newTrack->set_pz(pz);
491  newTrack->set_x(x);
492  newTrack->set_y(y);
493  newTrack->set_z(z);
494  newTrack->set_dca3d_xy(DCA3Dxy / Acts::UnitConstants::cm);
495  newTrack->set_dca3d_z(DCA3Dz / Acts::UnitConstants::cm);
496  newTrack->set_dca3d_xy_error(DCA3DxyCov / Acts::UnitConstants::cm);
497  newTrack->set_dca3d_z_error(DCA3DzCov / Acts::UnitConstants::cm);
498 
499  rotater->setVerbosity(0);
500  rotater->fillSvtxTrackStates(traj, trackTip,
501  dynamic_cast<SvtxTrack*>(newTrack),
502  m_tGeometry->geoContext,
503  m_hitIdClusKey);
504 
505  for(int i = 0; i < 6; ++i)
506  {
507  for(int j = 0; j < 6; ++j)
508  {
509  newTrack->set_error(i,j, rotatedCov(i,j));
510  }
511  }
512 
513 
514  trackTipTrackKeyMap.insert(std::pair<const size_t, const unsigned int>
515  (trackTip, lastTrackKey));
516 
517  m_trackMap->insert(newTrack);
518  }
519 
520  ++iTrack;
521  }
522 
524  m_actsTrackKeyMap->insert(std::pair<const unsigned int,
525  std::map<const size_t, const unsigned int>>
526  (trackKey, trackTipTrackKeyMap));
