98  void midPointAttachZ(VolumeConfig& lConfig) {
99  if (lConfig.zMin >= zMax) {
100  double zMid = 0.5 * (lConfig.zMin + zMax);
101  lConfig.zMin = zMid;
102  zMax = zMid;
103  } else {
104  double zMid = 0.5 * (zMin + lConfig.zMax);
105  lConfig.zMax = zMid;
106  zMin = zMid;
107  }
108  }

114  void attachZ(const VolumeConfig& lConfig) {
115  if (lConfig.zMin >= zMax) {
116  zMax = lConfig.zMin;
117  } else {
118  zMin = lConfig.zMax;
119  }
120  }

126  bool overlapsInR(const VolumeConfig& vConfig) const {
127  if (!present) {
128  return false;
129  }
130  return std::max(rMin, vConfig.rMin) <= std::min(rMax, vConfig.rMax);

137  bool overlapsInZ(const VolumeConfig& vConfig) const {
138  if (!present) {

148  bool wraps(const VolumeConfig& vConfig) const {
149  if ((zMax <= vConfig.zMin) || (zMin >= vConfig.zMax)) {
150  return true;
151  }
152  return containsInR(vConfig);

216  void configureContainerVolume() {
217  // set the container to be present
218  containerVolumeConfig.present = true;
219  std::string wConditionAddon = "";
220  // if we have more than one config present
221  if ((nVolumeConfig && cVolumeConfig) || (cVolumeConfig && pVolumeConfig) ||
222  (nVolumeConfig && pVolumeConfig)) {
223  wCondition = Wrapping;
224  wConditionScreen = "grouped to ";
225  }
226  // adapt the new volume config to the existing configs
227  if (nVolumeConfig) {
228  containerVolumeConfig.adapt(nVolumeConfig);
229  wConditionScreen += "[n]";
230  }
231  if (cVolumeConfig) {
232  containerVolumeConfig.adapt(cVolumeConfig);
233  wConditionScreen += "[c]";
234  }
235  if (pVolumeConfig) {
236  containerVolumeConfig.adapt(pVolumeConfig);
237  wConditionScreen += "[p]";
238  }
239  // adapt the external one
240  if (externalVolumeConfig) {
241  containerVolumeConfig.adapt(externalVolumeConfig);
242  }
243  // attach the volume configs
244  if (nVolumeConfig && cVolumeConfig) {
245  nVolumeConfig.midPointAttachZ(cVolumeConfig);
246  }
247  if (cVolumeConfig && pVolumeConfig) {
248  cVolumeConfig.midPointAttachZ(pVolumeConfig);
249  }
250  // adapt r afterwards
251  // - easy if no exisitng volume
252  // - possible if no central volume
253  if (!existingVolumeConfig || !cVolumeConfig) {
254  nVolumeConfig.adaptR(containerVolumeConfig);
255  cVolumeConfig.adaptR(containerVolumeConfig);
256  pVolumeConfig.adaptR(containerVolumeConfig);
257  }
258  }

261  void wrapInsertAttach() {
262  // action is only needed if an existing volume
263  // is present
264  if (existingVolumeConfig) {
265  // 0 - simple attachment case
266  if (!cVolumeConfig) {
267  // check if it can be easily attached
268  if (nVolumeConfig && nVolumeConfig.zMax < existingVolumeConfig.zMin) {
269  nVolumeConfig.attachZ(existingVolumeConfig);
270  // will attach the new volume(s)
271  wCondition = Attaching;
272  wConditionScreen = "[n attched]";
273  }
274  if (pVolumeConfig && pVolumeConfig.zMin > existingVolumeConfig.zMax) {
275  pVolumeConfig.attachZ(existingVolumeConfig);
276  // will attach the new volume(s)
277  wCondition = Attaching;
278  wConditionScreen = "[p attched]";
279  }
280  // see if inner glue volumes are needed
281  if (containerVolumeConfig.rMin > existingVolumeConfig.rMin) {
282  nVolumeConfig.rMin = existingVolumeConfig.rMin;
283  pVolumeConfig.rMin = existingVolumeConfig.rMin;
284  } else {
285  fGapVolumeConfig.present = true;
286  // get the zMin/zMax boundaries
287  fGapVolumeConfig.adaptZ(existingVolumeConfig);
288  fGapVolumeConfig.rMin = containerVolumeConfig.rMin;
289  fGapVolumeConfig.rMax = existingVolumeConfig.rMin;
290  }
291  // see if outer glue volumes are needed
292  if (containerVolumeConfig.rMax < existingVolumeConfig.rMax) {
293  nVolumeConfig.rMax = existingVolumeConfig.rMax;
294  pVolumeConfig.rMax = existingVolumeConfig.rMax;
295  } else {
296  sGapVolumeConfig.present = true;
297  // get the zMin/zMax boundaries
298  sGapVolumeConfig.adaptZ(existingVolumeConfig);
299  sGapVolumeConfig.rMin = existingVolumeConfig.rMax;
300  sGapVolumeConfig.rMax = containerVolumeConfig.rMax;
301  }
302  } else {
303  // full wrapping or full insertion case
304  if (existingVolumeConfig.rMax < containerVolumeConfig.rMin) {
305  // Full wrapping case
306  // - set the rMin
307  nVolumeConfig.rMin = existingVolumeConfig.rMax;
308  cVolumeConfig.rMin = existingVolumeConfig.rMax;
309  pVolumeConfig.rMin = existingVolumeConfig.rMax;
310  // - set the rMax
311  nVolumeConfig.rMax = containerVolumeConfig.rMax;
312  cVolumeConfig.rMax = containerVolumeConfig.rMax;
313  pVolumeConfig.rMax = containerVolumeConfig.rMax;
314  // will wrap the new volume(s) around existing
315  wCondition = Wrapping;
316  wConditionScreen = "[fully wrapped]";
317  } else if (existingVolumeConfig.rMin > containerVolumeConfig.rMax) {
318  // full insertion case
319  // set the rMax
320  nVolumeConfig.rMax = existingVolumeConfig.rMin;
321  cVolumeConfig.rMax = existingVolumeConfig.rMin;
322  pVolumeConfig.rMax = existingVolumeConfig.rMin;
323  // set the rMin
324  nVolumeConfig.rMin = containerVolumeConfig.rMin;
325  cVolumeConfig.rMin = containerVolumeConfig.rMin;
326  pVolumeConfig.rMin = containerVolumeConfig.rMin;
327  // will insert the new volume(s) into existing
328  wCondition = Inserting;
329  wConditionScreen = "[fully inserted]";
330  } else if (cVolumeConfig.wraps(existingVolumeConfig)) {
331  // central wrapping case
332  // set the rMax
333  nVolumeConfig.rMax = containerVolumeConfig.rMax;
334  cVolumeConfig.rMax = containerVolumeConfig.rMax;
335  pVolumeConfig.rMax = containerVolumeConfig.rMax;
336  // set the rMin
337  nVolumeConfig.rMin = existingVolumeConfig.rMin;
338  cVolumeConfig.rMin = existingVolumeConfig.rMax;
339  pVolumeConfig.rMin = existingVolumeConfig.rMin;
340  // set the Central Wrapping
341  wCondition = CentralWrapping;
342  wConditionScreen = "[centrally wrapped]";
343  } else if (existingVolumeConfig.wraps(cVolumeConfig)) {
344  // central insertion case
345  // set the rMax
346  nVolumeConfig.rMax = containerVolumeConfig.rMax;
347  cVolumeConfig.rMax = existingVolumeConfig.rMin;
348  pVolumeConfig.rMax = containerVolumeConfig.rMax;
349  // set the rMin
350  nVolumeConfig.rMin = containerVolumeConfig.rMin;
351  cVolumeConfig.rMin = containerVolumeConfig.rMin;
352  pVolumeConfig.rMin = containerVolumeConfig.rMin;
353  // set the Central Wrapping
354  wCondition = CentralWrapping;
355  wConditionScreen = "[centrally inserted]";
356  }
357 
358  // check if gaps are needed
359  //
360  // the gap reference is either the container for FULL wrapping,
361  // insertion
362  // or it is the centralVolume for central wrapping, insertion
363  VolumeConfig referenceVolume =
364  (wCondition == Wrapping || wCondition == Inserting)
365  ? containerVolumeConfig
366  : cVolumeConfig;
367  // - at the negative sector
368  if (existingVolumeConfig.zMin > referenceVolume.zMin) {
369  fGapVolumeConfig.present = true;
370  fGapVolumeConfig.adaptR(existingVolumeConfig);
371  fGapVolumeConfig.zMin = referenceVolume.zMin;
372  fGapVolumeConfig.zMax = existingVolumeConfig.zMin;
373  } else {
374  // adapt lower z boundary
375  if (nVolumeConfig) {
376  nVolumeConfig.zMin = existingVolumeConfig.zMin;
377  } else if (cVolumeConfig) {
378  cVolumeConfig.zMin = existingVolumeConfig.zMin;
379  }
380  }
381  // - at the positive sector
382  if (existingVolumeConfig.zMax < referenceVolume.zMax) {
383  sGapVolumeConfig.present = true;
384  sGapVolumeConfig.adaptR(existingVolumeConfig);
385  sGapVolumeConfig.zMin = existingVolumeConfig.zMax;
386  sGapVolumeConfig.zMax = referenceVolume.zMax;
387  } else {
388  // adapt higher z boundary
389  if (pVolumeConfig) {
390  pVolumeConfig.zMax = existingVolumeConfig.zMax;
391  } else if (cVolumeConfig) {
392  cVolumeConfig.zMax = existingVolumeConfig.zMax;
393  }
394  }
395  }
396  }
397  return;
398  }
