236 void KFParticle_truthAndDetTools::fillDetectorBranch(PHCompositeNode *topNode,
237  TTree *m_tree, KFParticle daughter, int daughter_id)
238 {
239  PHNodeIterator nodeIter(topNode);
240 
241  PHNode *findNode = dynamic_cast<PHNode*>(nodeIter.findFirst("SvtxTrackMap"));
242  if (findNode)
243  {
244  dst_trackmap = findNode::getClass<SvtxTrackMap>(topNode, "SvtxTrackMap");
245  }
246  else
247  {
248  std::cout << "KFParticle detector info: SvtxTrackMap does not exist" << std::endl;
249  }
250 
251  findNode = dynamic_cast<PHNode*>(nodeIter.findFirst("TRKR_CLUSTER"));
252  if (findNode)
253  {
254  dst_clustermap = findNode::getClass<TrkrClusterContainer>(topNode, "TRKR_CLUSTER");
255  }
256  else
257  {
258  std::cout << "KFParticle detector info: TRKR_CLUSTER does not exist" << std::endl;
259  }
260 
261  track = getTrack(daughter.Id(), dst_trackmap);
262 
263  for (SvtxTrack::ConstClusterKeyIter iter = track->begin_cluster_keys();
264  iter != track->end_cluster_keys();
265  ++iter)
266  {
267  TrkrDefs::cluskey clusKey = *iter;
268  TrkrCluster *cluster = dst_clustermap->findCluster(clusKey);
269  const unsigned int trkrId = TrkrDefs::getTrkrId(clusKey);
270 
271  detector_local_x[daughter_id].push_back(cluster->getX());
272  detector_local_y[daughter_id].push_back(cluster->getY());
273  detector_local_z[daughter_id].push_back(cluster->getZ());
274  detector_layer[daughter_id].push_back(TrkrDefs::getLayer(clusKey));
275  unsigned int staveId, chipId, ladderZId, ladderPhiId, sectorId, side;
276  staveId = chipId = ladderZId = ladderPhiId = sectorId = side = -99;
277 
278  if (Use["MVTX"] && trkrId == TrkrDefs::mvtxId)
279  {
280  staveId = MvtxDefs::getStaveId(clusKey);
281  chipId = MvtxDefs::getChipId(clusKey);
282  }
283  else if (Use["INTT"] && trkrId == TrkrDefs::inttId)
284  {
285  ladderZId = InttDefs::getLadderZId(clusKey);
286  ladderPhiId = InttDefs::getLadderPhiId(clusKey);
287  }
288  else if (Use["TPC"] && trkrId == TrkrDefs::tpcId)
289  {
290  sectorId = TpcDefs::getSectorId(clusKey);
291  side = TpcDefs::getSide(clusKey);
292  }
293 
294  mvtx_staveID[daughter_id].push_back(staveId);
295  mvtx_chipID[daughter_id].push_back(chipId);
296  intt_ladderZID[daughter_id].push_back(ladderZId);
297  intt_ladderPhiID[daughter_id].push_back(ladderPhiId);
298  tpc_sectorID[daughter_id].push_back(sectorId);
299  tpc_side[daughter_id].push_back(side);
300  }
