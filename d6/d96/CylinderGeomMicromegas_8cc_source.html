55  // check radius
56  if( !check_radius(world_location) ) return std::make_pair(-1,-1);

152 TVector3 CylinderGeomMicromegas::get_world_coordinate( uint tileid, uint stripnum ) const
153 {
154  assert( tileid < m_tiles.size() );
155 
156  // get tile
157  const auto& tile = m_tiles[tileid];
158 
159  switch( m_segmentation_type )
160  {
161  case MicromegasDefs::SegmentationType::SEGMENTATION_PHI:
162  {
163  const double z = tile.m_centerZ;
164  const double phi = tile.m_centerPhi - tile.m_sizePhi/2 + (0.5+stripnum)*m_pitch/m_radius;
165  return TVector3( m_radius*std::cos(phi), m_radius*std::sin(phi), z );
166  }
167 
168  case MicromegasDefs::SegmentationType::SEGMENTATION_Z:
169  {
170  const double z = tile.m_centerZ - tile.m_sizeZ/2 + (0.5+stripnum)*m_pitch;
171  const double phi = tile.m_centerPhi;
172  return TVector3( m_radius*std::cos(phi), m_radius*std::sin(phi), z );
173  }

175  }
