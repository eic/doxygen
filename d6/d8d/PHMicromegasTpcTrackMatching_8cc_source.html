372 void PHMicromegasTpcTrackMatching::CircleFitByTaubin (std::vector<TrkrCluster*> clusters, double &R, double &X0, double &Y0)
373 /*
374  Circle fit to a given set of data points (in 2D)
375  This is an algebraic fit, due to Taubin, based on the journal article
376  G. Taubin, "Estimation Of Planar Curves, Surfaces And Nonplanar
377  Space Curves Defined By Implicit Equations, With
378  Applications To Edge And Range Image Segmentation",
379  IEEE Trans. PAMI, Vol. 13, pages 1115-1138, (1991)
380  It works well whether data points are sampled along an entire circle or along a small arc.
381  It still has a small bias and its statistical accuracy is slightly lower than that of the geometric fit (minimizing geometric distances),
382  It provides a very good initial guess for a subsequent geometric fit.
383  Nikolai Chernov (September 2012)
384 */
385 {
386  int iter,IterMAX=99;
387 
388  double Mz,Mxy,Mxx,Myy,Mxz,Myz,Mzz,Cov_xy,Var_z;
389  double A0,A1,A2,A22,A3,A33;
390  double x,y;
391  double DET,Xcenter,Ycenter;
392 
393  // Compute x- and y- sample means
394  double meanX = 0;
395  double meanY = 0;
396  double weight = 0;
397  for(unsigned int iclus = 0; iclus < clusters.size(); ++iclus)
398  {
399  meanX += clusters[iclus]->getX();
400  meanY += clusters[iclus]->getY();
401  weight++;
402  }
403  meanX /= weight;
404  meanY /= weight;
405 
406  // computing moments
407 
408  Mxx=Myy=Mxy=Mxz=Myz=Mzz=0.;
409 
410  for (unsigned int i=0; i<clusters.size(); i++)
411  {
412  double Xi = clusters[i]->getX() - meanX; // centered x-coordinates
413  double Yi = clusters[i]->getY() - meanY; // centered y-coordinates
414  double Zi = Xi*Xi + Yi*Yi;
415 
416  Mxy += Xi*Yi;
417  Mxx += Xi*Xi;
418  Myy += Yi*Yi;
419  Mxz += Xi*Zi;
420  Myz += Yi*Zi;
421  Mzz += Zi*Zi;
422  }
423  Mxx /= weight;
424  Myy /= weight;
425  Mxy /= weight;
426  Mxz /= weight;
427  Myz /= weight;
428  Mzz /= weight;
429 
430  // computing coefficients of the characteristic polynomial
431 
432  Mz = Mxx + Myy;
433  Cov_xy = Mxx*Myy - Mxy*Mxy;
434  Var_z = Mzz - Mz*Mz;
435  A3 = 4*Mz;
436  A2 = -3*Mz*Mz - Mzz;
437  A1 = Var_z*Mz + 4*Cov_xy*Mz - Mxz*Mxz - Myz*Myz;
438  A0 = Mxz*(Mxz*Myy - Myz*Mxy) + Myz*(Myz*Mxx - Mxz*Mxy) - Var_z*Cov_xy;
439  A22 = A2 + A2;
440  A33 = A3 + A3 + A3;
441 
442  // finding the root of the characteristic polynomial
443  // using Newton's method starting at x=0
444  // (it is guaranteed to converge to the right root)
445 
446  for (x=0.,y=A0,iter=0; iter<IterMAX; iter++) // usually, 4-6 iterations are enough
447  {
448  double Dy = A1 + x*(A22 + A33*x);
449  double xnew = x - y/Dy;
450  if ((xnew == x)||(!isfinite(xnew))) break;
451  double ynew = A0 + xnew*(A1 + xnew*(A2 + xnew*A3));
452  if (fabs(ynew)>=fabs(y)) break;
453  x = xnew; y = ynew;
454  }
455 
456  // computing parameters of the fitting circle
457 
458  DET = x*x - x*Mz + Cov_xy;
459  Xcenter = (Mxz*(Myy - x) - Myz*Mxy)/DET/2;
460  Ycenter = (Myz*(Mxx - x) - Mxz*Mxy)/DET/2;
461 
462  // assembling the output
463 
464  X0 = Xcenter + meanX;
465  Y0 = Ycenter + meanY;
466  R = sqrt(Xcenter*Xcenter + Ycenter*Ycenter + Mz);
467 }

469 void PHMicromegasTpcTrackMatching::circle_circle_intersection(double r1, double r2, double x2, double y2, double &xplus, double &yplus, double &xminus, double &yminus)
470 {
471  // r1 is radius of sPHENIX layer
472  // r2, x2 and y2 are parameters of circle fitted to TPC clusters
473  // the solutions are xplus, xminus, yplus, yminus
474 
475  // The intersection of two circles occurs when
476  // (x-x1)^2 + (y-y1)^2 = r1^2, / (x-x2)^2 + (y-y2)^2 = r2^2
477  // Here we assume that circle 1 is an sPHENIX layer centered on x1=y1=0, and circle 2 is arbitrary
478  // x^2 +y^2 = r1^2, (x-x2)^2 + (y-y2)^2 = r2^2
479  // expand the equations and subtract to eliminate the x^2 and y^2 terms, gives the radical line connecting the intersection points
480  // iy = - (2*x2*x - D) / 2*y2,
481  // then substitute for y in equation of circle 1
482 
483  double D = r1*r1 - r2*r2 + x2*x2 + y2*y2;
484  double a = 1.0 + (x2*x2) / (y2*y2);
485  double b = - D * x2/( y2*y2);
486  double c = D*D / (4.0*y2*y2) - r1*r1;
487 
488  xplus = (-b + sqrt(b*b - 4.0* a * c) ) / (2.0 * a);
489  xminus = (-b - sqrt(b*b - 4.0* a * c) ) / (2.0 * a);
490 
491  // both values of x are valid
492  // but for each of those values, there are two possible y values on circle 1
493  // but only one of those falls on the radical line:
494 
495  yplus = - (2*x2*xplus - D) / (2.0*y2);
496  yminus = -(2*x2*xminus - D) / (2.0*y2);
497 
498 }

500 void PHMicromegasTpcTrackMatching::line_fit(std::vector<TrkrCluster*> clusters, double &a, double &b)
501 {
502  // copied from: https://www.bragitoff.com
503  // we want to fit z vs radius
504 
505  double xsum=0,x2sum=0,ysum=0,xysum=0; //variables for sums/sigma of xi,yi,xi^2,xiyi etc
506  for (unsigned int i=0; i<clusters.size(); ++i)
507  {
508  double z = clusters[i]->getZ();
509  double r = sqrt(pow(clusters[i]->getX(),2) + pow(clusters[i]->getY(), 2));
510 
511  xsum=xsum+r; //calculate sigma(xi)
512  ysum=ysum+z; //calculate sigma(yi)
513  x2sum=x2sum+pow(r,2); //calculate sigma(x^2i)
514  xysum=xysum+r*z; //calculate sigma(xi*yi)
515  }
516  a=(clusters.size()*xysum-xsum*ysum)/(clusters.size()*x2sum-xsum*xsum); //calculate slope
517  b=(x2sum*ysum-xsum*xysum)/(x2sum*clusters.size()-xsum*xsum); //calculate intercept
518 
519  if(Verbosity() > 10)
