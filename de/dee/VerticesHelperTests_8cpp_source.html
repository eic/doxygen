38  auto trfSpace = [](std::vector<Vector3D>& vtxs,
39  const Transform3D& trf) -> void {
40  std::transform(vtxs.begin(), vtxs.end(), vtxs.begin(),
41  [&](auto& v) { return (trf * v); });
42  };
