44  }
45 
46  // FIXME: suitable for 2D & 3D parameter space only;
47  if (tree->GetVerbosityLevel() >= 1)
48  printf("lv=%2d -> accu=%5d/%5d/%5d; mCellSize[]: %8.3f %8.3f %8.3f\n",

52 
53  mTotalCellNum = (ldim == 0 ? 1 : tree->GetLevel(ldim-1)->mTotalCellNum)*mDaughterCellNum;
54

58 
59  {
60  // In 4D case id(a,b,c,d) = a*d2*d3*d4 + b*d3*d4 + c*d4 + d; sdim[] is just

63
