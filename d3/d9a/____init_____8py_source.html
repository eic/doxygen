48 def find_root_lib_dir():
49  '''
50  Attempt to locate the ROOT library directory.
51  Return the path to it, or raise an exception if it cannot be found.
52  '''
53  try:
54  rootlib = ''
55  # First check with the ROOTSYS environment variable.
56  rootsys = os.path.expandvars('$ROOTSYS')
57  if os.path.exists(rootsys):
58  # This will give the ROOT top-level directory, while
59  # we want the lib subdirectory.
60  rootlib = '/'.join([rootsys, 'lib'])
61  else:
62  # If that's empty, attempt to locate via the location
63  # of the ROOT binary.
64  # If 'which root' fails a CalledProcessError is raised.
65  rootbin = subprocess.check_output(['which', 'root'])
66  # rootbin should be of the form '$ROOTSYS/bin/root\n'.
67  # We want the corresponding lib directory.
68  rootlib = rootbin.replace('bin/root\n', 'lib')
69  # Check that the ROOT library directory we found does
70  # indeed exist. If not, raise an exception.
71  if not os.path.exists(rootlib):
72  raise IOError(
73  ' '.join(['ROOT lib directory', rootlib, 'does not exist']))
74  # Check that the lib directory contains the Python library.
75  libpyroot = '/'.join([rootlib, 'libPyROOT.so'])
76  if not os.path.exists(libpyroot):
77  raise IOError('PyROOT library not found at ' + libpyroot)
78  return rootlib
79  # If 'which root' fails, catch the exception and re-raise it,
80  # printing something more informative.
81  except subprocess.CalledProcessError as e:
82  print 'Failed to locate ROOT; is it installed?'
