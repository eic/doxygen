210 void ActsTransformations::calculateDCA(const Acts::BoundTrackParameters param,

213  Acts::GeometryContext geoCtxt,

218 {
219  Acts::Vector3D pos = param.position(geoCtxt);
220  Acts::Vector3D mom = param.momentum();
221 
223  pos -= vertex;
224 
225  Acts::ActsSymMatrixD<3> posCov;
226  for(int i = 0; i < 3; ++i)
227  {
228  for(int j = 0; j < 3; ++j)
229  {
230  posCov(i,j) = cov(i,j);
231  }
232  }

243  rot(1,1) = cos(phi);
244  rot(1,2) = 0;
245  rot(2,0) = 0;
246  rot(2,1) = 0;
247  rot(2,2) = 1;
