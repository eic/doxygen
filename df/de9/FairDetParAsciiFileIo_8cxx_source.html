92 void FairDetParAsciiFileIo::readComment(const Char_t* lastBuf, FairParSet* pPar)
93 {
94  // reads the author or the description
95  const Int_t maxbuf=4000;
96  Text_t buf[maxbuf];
97  TString s=lastBuf;
98  TString pName, pVal;
99  Ssiz_t n=s.First(':');
100  Ssiz_t m=s.Last('\\');
101  pName=s(0,n);
102  if (m>0) { pVal=s(n+1,m-n-1); }
103  else { pVal=s(n+1,s.Length()-n-1); }
104  pVal=pVal.Strip(pVal.kLeading);
105  while (m>0) {
106  pVal+="\n";
107  pFile->getline(buf,maxbuf);
108  if (buf[0]!='/') {
109  s=buf;
110  m=s.Last('\\');
111  if (m>0) { pVal+=s(0,m); }
112  else { pVal+=s; }
113  }
114  }
115  if (pName.CompareTo("author")==0) { pPar->setAuthor(pVal.Data()); }
116  else if ( pName.CompareTo("description")==0) { pPar->setDescription(pVal.Data()); }
117 }
