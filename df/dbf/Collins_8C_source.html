187

333 {
334  gSystem->Load("libDelphes");
335 
336  TChain *chain = new TChain("Delphes");
337  chain->Add(inputFile);
338

340  ExRootResult *result = new ExRootResult();
341 
342  TestPlots *plots = new TestPlots;
343 
344  BookHistograms(result, plots);
345 
346  AnalyseEvents(treeReader, plots);
347 
348  PrintHistograms(result, plots);
349 
350  result->Write("results.root");
351 
352  cout << "** Exiting..." << endl;
353 
354  delete plots;
355  delete result;
356  delete treeReader;
357  delete chain;
