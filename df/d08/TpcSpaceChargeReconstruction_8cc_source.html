186 bool TpcSpaceChargeReconstruction::accept_track( SvtxTrack* track ) const
187 {
188  // ignore tracks whose transverse momentum is too small
189  const auto pt = std::sqrt( square( track->get_px() ) + square( track->get_py() ) );
190  if( pt < 0.5 ) return false;
191 
192  // ignore tracks with too few mvtx, intt and micromegas hits
193  if( get_clusters<TrkrDefs::mvtxId>(track) < 2 ) return false;
194  if( get_clusters<TrkrDefs::inttId>(track) < 2 ) return false;
195  if( m_use_micromegas && get_clusters<TrkrDefs::micromegasId>(track) < 2 ) return false;
196 
197  // all tests passed
198  return true;

202 void TpcSpaceChargeReconstruction::process_track( SvtxTrack* track )
203 {
204 
205  // running track state
206  auto state_iter = track->begin_states();
207 
208  // loop over clusters
209  for( auto key_iter = track->begin_cluster_keys(); key_iter != track->end_cluster_keys(); ++key_iter )
210  {
211 
212  const auto& cluster_key = *key_iter;
213  auto cluster = m_cluster_map->findCluster( cluster_key );
214  if( !cluster )
215  {
216  std::cout << PHWHERE << " unable to find cluster for key " << cluster_key << std::endl;
217  continue;
218  }
219 
220  ++m_total_clusters;
221 
222  // make sure
223  const auto detId = TrkrDefs::getTrkrId(cluster_key);
224  if(detId != TrkrDefs::tpcId) continue;
225 
226  // cluster r, phi and z
227  const auto cluster_r = get_r( cluster->getX(), cluster->getY() );
228  const auto cluster_phi = std::atan2( cluster->getY(), cluster->getX() );
229  const auto cluster_z = cluster->getZ();
230 
231  // cluster errors
232  const auto cluster_rphi_error = cluster->getRPhiError();
233  const auto cluster_z_error = cluster->getZError();
234 
235  /*
236  remove clusters with too small errors since they are likely pathological
237  and have a large contribution to the chisquare
238  TODO: make these cuts configurable
239  */
240  if( cluster_rphi_error < 0.015 ) continue;
241  if( cluster_z_error < 0.05 ) continue;
242 
243  // find track state that is the closest to cluster
244  /* this assumes that both clusters and states are sorted along r */
245  float dr_min = -1;
246  for( auto iter = state_iter; iter != track->end_states(); ++iter )
247  {
248  const auto dr = std::abs( cluster_r - get_r( iter->second->get_x(), iter->second->get_y() ) );
249  if( dr_min < 0 || dr < dr_min )
250  {
251  state_iter = iter;
252  dr_min = dr;
253  } else break;
254  }
255 
256  // get relevant track state
257  const auto state = state_iter->second;
258 
259  // extrapolate track parameters to the cluster r
260  const auto track_r = get_r( state->get_x(), state->get_y() );
261  const auto dr = cluster_r - track_r;
262  const auto track_drdt = (state->get_x()*state->get_px() + state->get_y()*state->get_py())/track_r;
263  const auto track_dxdr = state->get_px()/track_drdt;
264  const auto track_dydr = state->get_py()/track_drdt;
265  const auto track_dzdr = state->get_pz()/track_drdt;
266 
267  // store state position
268  const auto track_x = state->get_x() + dr*track_dxdr;
269  const auto track_y = state->get_y() + dr*track_dydr;
270  const auto track_z = state->get_z() + dr*track_dzdr;
271  const auto track_phi = std::atan2( track_y, track_x );
272 
273  // get track angles
274  const auto cosphi( std::cos( track_phi ) );
275  const auto sinphi( std::sin( track_phi ) );
276  const auto track_pphi = -state->get_px()*sinphi + state->get_py()*cosphi;
277  const auto track_pr = state->get_px()*cosphi + state->get_py()*sinphi;
278  const auto track_pz = state->get_pz();
279  const auto talpha = -track_pphi/track_pr;
280  const auto tbeta = -track_pz/track_pr;
281 
282  // sanity check
283  if( std::isnan(talpha) )
284  {
285  std::cout << "TpcSpaceChargeReconstruction::process_track - talpha nan" << std::endl;
286  continue;
287  }
288 
289  if( std::isnan(tbeta) )
290  {
291  std::cout << "TpcSpaceChargeReconstruction::process_track - tbeta nan" << std::endl;
292  continue;
293  }
294 
295  // check against limits
296  if( std::abs( talpha ) > m_max_talpha ) continue;
297  if( std::abs( tbeta ) > m_max_tbeta ) continue;
298 
299  // track errors
300  const auto track_rphi_error = state->get_rphi_error();
301  const auto track_z_error = state->get_z_error();
302 
303  // residuals
304  const auto drp = cluster_r*delta_phi( cluster_phi - track_phi );
305  const auto dz = cluster_z - track_z;
306 
307  // sanity checks
308  if( std::isnan(drp) )
309  {
310  std::cout << "TpcSpaceChargeReconstruction::process_track - drp nan" << std::endl;
311  continue;
312  }
313 
314  if( std::isnan(dz) )
315  {
316  std::cout << "TpcSpaceChargeReconstruction::process_track - dz nan" << std::endl;
317  continue;
318  }
319 
320  // check against limits
321  if( std::abs( drp ) > m_max_drphi ) continue;
322  if( std::abs( dz ) > m_max_dz ) continue;
323 
324  // residual errors squared
325  const auto erp = square(track_rphi_error) + square(cluster_rphi_error);
326  const auto ez = square(track_z_error) + square(cluster_z_error);
327 
328  // sanity check
329  // TODO: check whether this happens and fix upstream
330  if( std::isnan( erp ) )
331  {
332  std::cout << "TpcSpaceChargeReconstruction::process_track - erp nan" << std::endl;
333  continue;
334  }
335 
336  if( std::isnan( ez ) )
337  {
338  std::cout << "TpcSpaceChargeReconstruction::process_track - ez nan" << std::endl;
339  continue;
340  }
341 
342  // get cell
343  const auto i = get_cell( cluster );
344 
345  if( i < 0 || i >= m_totalbins ) continue;
346 
347  m_lhs[i](0,0) += 1./erp;
348  m_lhs[i](0,1) += 0;
349  m_lhs[i](0,2) += talpha/erp;
350 
351  m_lhs[i](1,0) += 0;
352  m_lhs[i](1,1) += 1./ez;
353  m_lhs[i](1,2) += tbeta/ez;
354 
355  m_lhs[i](2,0) += talpha/erp;
356  m_lhs[i](2,1) += tbeta/ez;
357  m_lhs[i](2,2) += square(talpha)/erp + square(tbeta)/ez;
358 
359  m_rhs[i](0,0) += drp/erp;
360  m_rhs[i](1,0) += dz/ez;
361  m_rhs[i](2,0) += talpha*drp/erp + tbeta*dz/ez;
362 
363  ++m_accepted_clusters;
364  ++m_cluster_count[i];
365 
366  }

487 int TpcSpaceChargeReconstruction::get_cell( float phi, float r, float z ) const
488 {
489 
490  // phi
491  // bound check
492  while( phi < m_phimin ) phi += 2.*M_PI;
493  while( phi >= m_phimax ) phi -= 2.*M_PI;
494  int iphi = m_phibins*(phi-m_phimin)/(m_phimax-m_phimin);
495 
496  // radius
497  if( r < m_rmin || r >= m_rmax ) return -1;
498  int ir = m_rbins*(r-m_rmin)/(m_rmax-m_rmin);
499 
500  // z
501  if( z < m_zmin || z >= m_zmax ) return -1;
502  int iz = m_zbins*(z-m_zmin)/(m_zmax-m_zmin);
503 
504  return get_cell( iphi, ir, iz );
505 }
