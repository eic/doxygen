35 std::vector<std::shared_ptr<Acts::TrackingVolume>>
36 Acts::DD4hepVolumeBuilder::centralVolumes() const {
37  if (m_cfg.centralVolumes.empty()) {
38  ACTS_VERBOSE("[L] No layers handed over for central volume!");
39  return {};
40  }
41 
42  ACTS_VERBOSE(
43  "[L] Received layers for central volume -> creating "
44  "cylindrical layers");
45 
46  // Resulting volumes
47  MutableTrackingVolumeVector volumes;
48  // Inner/outer radius and half length of the barrel
49  double rMin, rMax, dz;
50 
51  // Go through volumes
52  for (auto& detElement : m_cfg.centralVolumes) {
53  // Access the global transformation matrix of the volume
54  auto transform =
55  convertTransform(&(detElement.nominal().worldTransformation()));
56  // Get the shape of the volume
57  TGeoShape* geoShape = detElement.placement().ptr()->GetVolume()->GetShape();
58 
59  if (geoShape != nullptr) {
60  TGeoTubeSeg* tube = dynamic_cast<TGeoTubeSeg*>(geoShape);
61  if (tube == nullptr)
62  ACTS_ERROR(
63  "[L] Cylinder layer has wrong shape - needs to be TGeoTubeSeg!");
64 
65  // Extract the boundaries
66  rMin = tube->GetRmin() * UnitConstants::cm;
67  rMax = tube->GetRmax() * UnitConstants::cm;
68  dz = tube->GetDz() * UnitConstants::cm;
69 
70  } else {
71  throw std::logic_error(
72  std::string("Volume DetElement: ") + detElement.name() +
73  std::string(" has not a shape "
74  "added to its extension. Please check your detector "
75  "constructor!"));
76  }
77  // Build boundaries
78  CylinderVolumeBounds cvBounds(rMin, rMax, dz);
79  volumes.push_back(TrackingVolume::create(
80  transform, std::make_shared<const CylinderVolumeBounds>(cvBounds)));
81  }
82  return volumes;
83 }

85 Acts::Transform3D Acts::DD4hepVolumeBuilder::convertTransform(
86  const TGeoMatrix* tGeoTrans) const {
87  // Get the placement and orientation in respect to its mother
88  const Double_t* rotation = tGeoTrans->GetRotationMatrix();
89  const Double_t* translation = tGeoTrans->GetTranslation();
90  return TGeoPrimitivesHelper::makeTransform(
91  Acts::Vector3D(rotation[0], rotation[3], rotation[6]),
92  Acts::Vector3D(rotation[1], rotation[4], rotation[7]),
93  Acts::Vector3D(rotation[2], rotation[5], rotation[8]),
94  Acts::Vector3D(translation[0] * UnitConstants::cm,
95  translation[1] * UnitConstants::cm,
96  translation[2] * UnitConstants::cm));
97 }
