249 void PHTpcResiduals::calculateTpcResiduals(
250  const Acts::BoundTrackParameters &params,
251  const SourceLink& sl)
252 {
253  Acts::Vector2D local(sl.location()(0), sl.location()(1));
254 
255  auto globalSL = sl.referenceSurface().localToGlobal(
256  m_tGeometry->geoContext,
257  local,
258  Acts::Vector3D(1,1,1));
259 
261  const auto clusR = sqrt(pow(globalSL.x(), 2) +
262  pow(globalSL.y(), 2));
263  const auto clusPhi = std::atan2(globalSL.y(), globalSL.x());
264  const auto clusZ = globalSL.z();
265  const auto clusRPhiErr = sqrt(sl.covariance()(Acts::eBoundLoc0,
266  Acts::eBoundLoc0));
267  const auto clusZErr = sqrt(sl.covariance()(Acts::eBoundLoc1,
268  Acts::eBoundLoc1));
269 
270 
271  const auto globalStatePos = params.position(m_tGeometry->geoContext);
272  const auto globalStateCov = *params.covariance();
273  const auto stateRPhiErr = sqrt(globalStateCov(Acts::eBoundLoc0,
274  Acts::eBoundLoc0));
275  const auto stateZErr = sqrt(globalStateCov(Acts::eBoundLoc1,
276  Acts::eBoundLoc1));
277 
278  const auto statePhi = std::atan2(globalStatePos.y(),
279  globalStatePos.x());
280  const auto stateZ = globalStatePos.z();
281 
282  const auto erp = pow(stateRPhiErr, 2) + pow(clusRPhiErr, 2);
283  const auto ez = pow(stateZErr, 2) + pow(clusZErr, 2);
284 
285  const auto dPhi = clusPhi - statePhi;
286 
288  const auto drphi = clusR * deltaPhi(dPhi);
289  const auto dz = clusZ - stateZ;
290 
291  const auto trackEta
292  = std::atanh(params.momentum().z() / params.absoluteMomentum());
293  const auto clusEta = std::atanh(clusZ / globalSL.norm());
294 
295  h_rphiResid->Fill(clusR / Acts::UnitConstants::cm, drphi);
296  h_zResid->Fill(stateZ / Acts::UnitConstants::cm, dz);
297  h_etaResid->Fill(trackEta, clusEta - trackEta);
298  h_zResidLayer->Fill(clusR / Acts::UnitConstants::cm, dz);
299  h_etaResidLayer->Fill(clusR / Acts::UnitConstants::cm,
300  clusEta - trackEta);
301 
302  const auto trackPPhi = -params.momentum()(0) * std::sin(statePhi) +
303  params.momentum()(1) * std::cos(statePhi);
304  const auto trackPR = params.momentum()(0) * std::cos(statePhi) +
305  params.momentum()(1) * std::sin(statePhi);
306 
307  const auto trackPZ = params.momentum()(1);
308  const auto trackAlpha = -trackPPhi / trackPR;
309  const auto trackBeta = -trackPZ / trackPR;
310 
311  if(std::abs(trackAlpha) > m_maxTAlpha
312  || std::abs(drphi) > m_maxResidual)
313  return;
314 
315  const auto index = getCell(sl.referenceSurface().geometryId().layer(),
316  globalSL);
317 
318  if(index < 0 || index > m_totalBins)
319  return;
320 
322  m_lhs[index](0,0) += 1. / erp;
323  m_lhs[index](0,1) += 0;
324  m_lhs[index](0,2) += trackAlpha / erp;
325 
326  m_lhs[index](1,0) += 0;
327  m_lhs[index](1,1) += 1. / ez;
328  m_lhs[index](1,2) += trackBeta / ez;
329 
330  m_lhs[index](2,0) += trackAlpha / erp;
331  m_lhs[index](2,1) += trackBeta / ez;
332  m_lhs[index](2,2) += pow(trackAlpha, 2) / erp
333  + pow(trackBeta, 2) / ez;
334 
335  m_rhs[index](0,0) += drphi / erp;
336  m_rhs[index](1,0) += dz / ez;
337  m_rhs[index](2,0) += trackAlpha * drphi / erp + trackBeta * dz / ez;
338 
339  m_clusterCount[index]++;
340 
341  return;
342 }

344 void PHTpcResiduals::calculateDistortions(PHCompositeNode *topNode)
345 {
346  auto *geomContainer = findNode::getClass<PHG4CylinderCellGeomContainer>(topNode, "CYLINDERCELLGEOM_SVTX");
347  if(!geomContainer)
348  {
349  std::cout << PHWHERE << "No CYLINDERCELLGEOM_SVTX node, exiting."
350  << std::endl;
351  }
352 
353  std::vector<Acts::Vector3D> delta(m_totalBins);
354  std::vector<Acts::SymMatrix3D> cov(m_totalBins);
355 
356  for(int i = 0; i < m_totalBins; ++i)
357  {
358  if(m_clusterCount[i] < 10)
359  continue;
360  cov[i] = m_lhs[i].inverse();
361  delta[i] = m_lhs[i].partialPivLu().solve(m_rhs[i]);
362  }
363 
365  std::vector<std::unique_ptr<TGraphErrors>>
366  graphs(m_zBins * m_phiBins * m_nCoord);
367 
368  DistortionMap *distCorr = new DistortionMap;
369  DistortionMap *distCorrErr = new DistortionMap;
370 
371  for(int iz = 0; iz < m_zBins; ++iz) {
372  for(int iphi = 0; iphi < m_phiBins; ++iphi) {
373  for(int ir = 0; ir < m_rBins; ++ir) {
374  const int index = getCell(iz, ir, iphi);
375 
378  const int innerLayer = 7 + m_nLayersTpc *
379  ir / m_rBins;
380  const int outerLayer = 7 + m_nLayersTpc * (ir+1)
381  / m_rBins - 1;
382 
383  const auto innerRadius = geomContainer->
384  GetLayerCellGeom(innerLayer)->get_radius();
385  const auto outerRadius = geomContainer->
386  GetLayerCellGeom(outerLayer)->get_radius();
387  const float r = (innerRadius + outerRadius) / 2.;
388 
389  for(unsigned int icoord = 0;
390  icoord < m_nCoord; ++icoord) {
391  const int tgrIndex = iz + m_zBins *
392  ( iphi + m_phiBins * icoord);
393  graphs[tgrIndex].reset(new TGraphErrors());
394  graphs[tgrIndex]->SetName(Form("tg_%i_%i_%i", iz, iphi, icoord));
395 
398  if(m_clusterCount[index]<10)
399  {
401  distCorr->insert({index,
402  std::pair<unsigned int, const double>(0,0)});
403  distCorrErr->insert({index,
404  std::pair<unsigned int, const double>(0,0)});
405  continue;
406  }
407 
408  const double corrVal = delta[index](icoord,0);
409  const double corrValErr =
410  std::sqrt(cov[index](icoord, icoord));
411 
412  auto corr = std::pair<unsigned int,
413  const double>
414  (icoord, corrVal);
415  auto corrErr = std::pair<unsigned int,
416  const double>
417  (icoord, corrValErr);
418  distCorr->insert({index, corr});
419  distCorrErr->insert({index, corrErr});
420 
421  graphs[tgrIndex]->SetPoint(ir, r,
422  delta[index](icoord,0));
423  graphs[tgrIndex]->SetPointError(
424  ir, 0, std::sqrt(cov[index](icoord,icoord)));
425 
426  }
427  }
428  }
429  }
430 
433  m_distortionCorrections->m_distortionMap = distCorr;
434  m_distortionCorrections->m_distortionMapErr = distCorrErr;
435  m_distortionCorrections->m_zBins = m_zBins;
436  m_distortionCorrections->m_phiBins = m_phiBins;
437  m_distortionCorrections->m_rBins = m_rBins;
438  m_distortionCorrections->m_nCoord = m_nCoord;
439 
440 
442  TFile *outputFile =
443  new TFile((Name() + "_distortions.root").c_str(), "RECREATE");
444  outputFile->cd();
445  for(auto&& gr : graphs)
446  gr->Write();
447  outputFile->Close();
448

451 int PHTpcResiduals::getCell(const int actsLayer,
452  const Acts::Vector3D& loc)
453 {
456  const auto layer = (actsLayer / 2.) -1;
457  const int ir = m_rBins * layer / m_nLayersTpc;
458 
459  auto clusPhi = deltaPhi(std::atan2(loc(1), loc(0)));
460  const int iphi = m_phiBins * (clusPhi * M_PI) / (2. * M_PI);
461 
462  const int iz = m_zBins * (loc(2) - m_zMin) / (m_zMax - m_zMin);
463 
464  return getCell(iz, ir, iphi);
