83 int eventFileWriter::writeEvent(upcEvent &event, int eventnumber)
84 {
85 
86  int numberoftracks = 0;
87  if(_writeFullPythia)
88  {
89  numberoftracks = event.getParticles()->size();
90  }
91  else
92  {
93  for(unsigned int i = 0; i<event.getParticles()->size(); ++i)
94  {
95  if(event.getParticles()->at(i).getStatus() >= 0) numberoftracks++;
96  }
97  }
98 
99  // sometimes we don't have tracks due to wrongly picked W , check it
100  if(numberoftracks){
101  eventnumber++;
102 
103  _fileStream << "EVENT: " << eventnumber << " " << numberoftracks << " " << 1 << std::endl;
104  if(event.getGammaEnergies()->size()) _fileStream << "GAMMAENERGIES:";
105  for(unsigned int n = 0; n < event.getGammaEnergies()->size(); n++)
106  {
107  _fileStream << " " << event.getGammaEnergies()->at(n);
108  }
109  if(event.getGammaEnergies()->size()) _fileStream<< std::endl;
110  _fileStream <<"VERTEX: "<<0.<<" "<<0.<<" "<<0.<<" "<<0.<<" "<<1<<" "<<0<<" "<<0<<" "<<numberoftracks<<std::endl;
111 
112  int ipart = 0;
113  std::vector<starlightParticle>::const_iterator part = (event.getParticles())->begin();
114 
115  for (part = event.getParticles()->begin(); part != event.getParticles()->end(); part++, ipart++)
116  {
117  if(!_writeFullPythia)
118  {
119  if((*part).getStatus() < 0) continue;
120  }
121  _fileStream << "TRACK: " << " " << starlightParticleCodes::jetsetToGeant((*part).getPdgCode()) <<" "<< (*part).GetPx() << " " << (*part).GetPy()
122  << " "<< (*part).GetPz() << " " << eventnumber << " " << ipart << " " << 0 << " "
123  << (*part).getPdgCode();
124 
125  if(_writeFullPythia)
126  {
127  lorentzVector vtx = (*part).getVertex();
128  _fileStream << " " << vtx.GetPx() << " " << vtx.GetPy() << " " << vtx.GetPz() << " " << vtx.GetE();
129  _fileStream << " " << (*part).getFirstParent() << " " << (*part).getLastParent() << " " << (*part).getFirstDaughter() << " " << (*part).getLastDaughter() << " " << (*part).getStatus();
130  }
131 
132  _fileStream <<std::endl;
133  }
134  }
135 
136  return 0;
137 }
