28 static Intersection intersectionEstimate(const Transform3D& transform,
29  const Vector3D& position,
30  const Vector3D& direction) {
31  // Get the matrix from the transform (faster access)
32  const auto& tMatrix = transform.matrix();
33  const Vector3D pnormal = tMatrix.block<3, 1>(0, 2).transpose();
34  const Vector3D pcenter = tMatrix.block<3, 1>(0, 3).transpose();
35  // It is solvable, so go on
36  double denom = direction.dot(pnormal);
37  if (denom != 0.0) {
38  // Translate that into a path
39  double path = (pnormal.dot((pcenter - position))) / (denom);
40  // Is valid hence either on surface or reachable
41  Intersection::Status status =
42  (path * path < s_onSurfaceTolerance * s_onSurfaceTolerance)
43  ? Intersection::Status::onSurface
44  : Intersection::Status::reachable;
45  // Return the intersection
46  return Intersection{(position + path * direction), path, status};
47  }
48  return Intersection();
49 }
