63  T rnd_gauss(T mean, T stddev)
64  {
65  T u, v, r, gauss;
66  do
67  {
68  u = 2 * ((double) rand() / (RAND_MAX)) - 1;
69  v = 2 * ((double) rand() / (RAND_MAX)) - 1;
70  r = u * u + v * v;

663  T t13 = mCosPhase * mCosPhase;

670  }

674  }

685  T t11 = mOrigin.y() * mOrigin.y();

697  4.0 * t11 * mOrigin.y() * mCurvature * t2 + 4.0 * t11 - 4.0 * t14 +
