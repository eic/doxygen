44  double m1=999999;

64  //cout<<"track1 ";point1.Print();
65  plane1.setNormal(dir1);
66  rep1->GFAbsTrackRep::extrapolate(plane1);
67  rep2->extrapolateToPoint(point1,point2,norm2);
