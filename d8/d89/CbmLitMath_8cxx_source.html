58  litfloat C5 = par->GetCovariance(5);
59 
60  litfloat norm = dxx * dyy - dxx * C5 - dyy * C0 + C0 * C5
61  - dxy * dxy + 2 * dxy * C1 - C1 * C1;
62  if (norm == 0.) { norm = 1e-10; }
63  return ((xmx * (dyy - C5) - ymy * (dxy - C1)) * xmx
