44  // Look for the scattered lepton in the event record.
45  // This is the first (only?) particle that matches the following:
46  // 1) pdg code equals that of incident lepton beam.
47  // 2) status code is 1 i.e. it's a stable/final-state particle.
48  // 3) the parent is track 1 or 2
49  const ParticleMC* EventSartre::ScatteredLepton() const {
50  // Look for the lepton beam to get the species.
51  // If we don't get it we can't find the scattered
52  // lepton so return NULL.
53  const VirtualParticle* beam = BeamLepton();
54  if (!beam) {
55  return NULL;
56  } // if
57  const int species = beam->Id().Code();
58  // Get the final state particles and search them for
59  // the scattered lepton.
60  std::vector<const VirtualParticle*> final;
61  FinalState(final);
62  std::vector<const VirtualParticle*>::const_iterator iter;
63  for (iter = final.begin(); iter != final.end(); ++iter) {
64  // We already know the particle is final state, so
65  // check its species and parent index.
66  if ( (*iter)->Id().Code() == species &&
67  ( (*iter)->GetParentIndex() == 1 || (*iter)->GetParentIndex() == 2 )
68  ) {
69  // Found it, cast to required particle type and return.
70  return static_cast<const ParticleMC*>(*iter);
71  } // if
72  } // for
73  // No luck, couldn't find the scattered lepton.
74  return nullptr;
75  }
