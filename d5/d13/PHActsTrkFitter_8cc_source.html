491 void PHActsTrkFitter::checkSurfaceVec(SurfacePtrVec &surfaces)
492 {
493  for(int i = 0; i < surfaces.size() - 1; i++)
494  {
495  auto surface = surfaces.at(i);
496  auto thisVolume = surface->geometryId().volume();
497  auto thisLayer = surface->geometryId().layer();
498 
499  auto nextSurface = surfaces.at(i+1);
500  auto nextVolume = nextSurface->geometryId().volume();
501  auto nextLayer = nextSurface->geometryId().layer();
502 
504  if(nextVolume == thisVolume)
505  {
506  if(nextLayer < thisLayer)
507  {
508  if(Verbosity() > 2)
509  std::cout << PHWHERE
510  << "Surface not in order... removing surface"
511  << surface->geometryId() << std::endl;
512  surfaces.erase(surfaces.begin() + i);
514  i--;
515  continue;
516  }
517  }
518  else
519  {
520  if(nextVolume < thisVolume)
521  {
522  if(Verbosity() > 2)
523  std::cout << PHWHERE
524  << "Volume not in order... removing surface"
525  << surface->geometryId() << std::endl;
526  surfaces.erase(surfaces.begin() + i);
528  i--;
529  continue;
530  }
531  }
532  }

536 void PHActsTrkFitter::updateSvtxTrack(Trajectory traj,
537  const unsigned int trackKey,
538  Acts::Vector3D vertex)
539 {
540  const auto &[trackTips, mj] = traj.trajectory();
542  auto &trackTip = trackTips.front();
543 
544  SvtxTrackMap::Iter trackIter = m_trackMap->find(trackKey);
545  SvtxTrack *track = trackIter->second;
546 
547  if(Verbosity() > 2)
548  {
549  std::cout << "Identify (proto) track before updating with acts results " << std::endl;
550  track->identify();
551  std::cout << " cluster keys size " << track->size_cluster_keys() << std::endl;
552  }
553 
554  // The number of associated clusters may have changed - start over
555  track->clear_states();
556  track->clear_cluster_keys();
557 
558  // create a state at pathlength = 0.0
559  // This state holds the track parameters, which will be updated below
560  float pathlength = 0.0;
561  SvtxTrackState_v1 out( pathlength);
562  out.set_x(0.0);
563  out.set_y(0.0);
564  out.set_z(0.0);
565  track->insert_state(&out);
566 
567  auto trajState =
568  Acts::MultiTrajectoryHelpers::trajectoryState(mj, trackTip);
569 
570  const auto& params = traj.trackParameters(trackTip);
571 
573  track->set_x(params.position(m_tGeometry->geoContext)(0)
574  / Acts::UnitConstants::cm);
575  track->set_y(params.position(m_tGeometry->geoContext)(1)
576  / Acts::UnitConstants::cm);
577  track->set_z(params.position(m_tGeometry->geoContext)(2)
578  / Acts::UnitConstants::cm);
579 
580  track->set_px(params.momentum()(0));
581  track->set_py(params.momentum()(1));
582  track->set_pz(params.momentum()(2));
583 
584  track->set_charge(params.charge());
585  track->set_chisq(trajState.chi2Sum);
586  track->set_ndf(trajState.NDF);
587 
588  auto rotater = std::make_unique<ActsTransformations>();
589  rotater->setVerbosity(Verbosity());
590 
591  float dca3Dxy = NAN;
592  float dca3Dz = NAN;
593  float dca3DxyCov = NAN;
594  float dca3DzCov = NAN;
595 
596  if(params.covariance())
597  {
598 
599  Acts::BoundSymMatrix rotatedCov =
600  rotater->rotateActsCovToSvtxTrack(params,
601  m_tGeometry->geoContext);
602 
603  for(int i = 0; i < 6; i++)
604  {
605  for(int j = 0; j < 6; j++)
606  {
607  track->set_error(i,j, rotatedCov(i,j));
608  }
609  }
610 
611 
612 
613  rotater->calculateDCA(params, vertex, rotatedCov,
614  m_tGeometry->geoContext,
615  dca3Dxy, dca3Dz,
616  dca3DxyCov, dca3DzCov);
617  }
618 
621  track->set_dca3d_xy(dca3Dxy / Acts::UnitConstants::cm);
622  track->set_dca3d_z(dca3Dz / Acts::UnitConstants::cm);
623 
626  track->set_dca3d_xy_error(sqrt(dca3DxyCov));
627  track->set_dca3d_z_error(sqrt(dca3DzCov));
628 
629  // Also need to update the state list and cluster ID list for all measurements associated with the acts track
630  // loop over acts track states, copy over to SvtxTrackStates, and add to SvtxTrack
631 
632  auto trackStateTimer = std::make_unique<PHTimer>("TrackStateTimer");
633  trackStateTimer->stop();
634  trackStateTimer->restart();
635 
636  if(m_fillSvtxTrackStates)
637  rotater->fillSvtxTrackStates(traj, trackTip, track,
638  m_tGeometry->geoContext,
639  m_hitIdClusKey);
640 
641  trackStateTimer->stop();
642  auto stateTime = trackStateTimer->get_accumulated_time();
643 
644  if(Verbosity() > 0)
645  std::cout << "PHActsTrkFitter update SvtxTrackStates time "
646  << stateTime << std::endl;
647 
648  if(m_timeAnalysis)
649  h_stateTime->Fill(stateTime);
650 
651  if(Verbosity() > 2)
652  {
653  std::cout << " Identify fitted track after updating track states:"
654  << std::endl;
655  track->identify();
656  std::cout << " cluster keys size " << track->size_cluster_keys()
657  << std::endl;
658  }
659 
660  return;
661 
662 }
