62 double beam::photonDensity(const double impactparameter,
63  const double photonEnergy) const
64 {
65  // function for the calculation of the "photon density".
66  // photonFlux = number of photons / (energy * area)
67  // assume beta = 1 and gamma >> 1, i.e. neglect the (1 / gamma^2) * K_0(x) term
68 
69  const double X
70  = (impactparameter * photonEnergy) / (_beamLorentzGamma * starlightConstants::hbarc);
71  if (X <= 0)
72  printWarn << "X = " << X << endl;
73  const double factor1 = (double(Z() * Z()) * starlightConstants::alpha) / (starlightConstants::pi * starlightConstants::pi);
74  const double factor2 = 1. / (photonEnergy * impactparameter * impactparameter);
75  const double bessel = bessel::dbesk1(X);
76  const double factor3 = X * X * bessel * bessel;
77 
78  return factor1 * factor2 * factor3;
79 }
