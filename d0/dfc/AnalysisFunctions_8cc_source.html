35 inline std::map<std::string, float> DISVariables(TClonesArray *branchParticle)
36 {
37  // four-momenta of proton, electron, virtual photon/Z^0/W^+-.
38  auto pProton = static_cast<GenParticle *>(branchParticle->At(0))->P4(); //these numbers 0 , 3, 5 are hardcoded in Pythia8
39  auto pleptonIn = static_cast<GenParticle *>(branchParticle->At(3))->P4();
40  auto pleptonOut = static_cast<GenParticle *>(branchParticle->At(5))->P4();
41  auto pPhoton = pleptonIn - pleptonOut;
42 
43  // Q2, W2, Bjorken x, y, nu.
44  float Q2 = -pPhoton.M2();
45  float W2 = (pProton + pPhoton).M2();
46  float x = Q2 / (2. * pProton.Dot(pPhoton));
47  float y = (pProton.Dot(pPhoton)) / (pProton.Dot(pleptonIn));
48 
49

51  dis_variables["Q2"] = Q2;
52  dis_variables["W2"] = W2;
53  dis_variables["x"] = x;
54  dis_variables["y"] = y;

59 inline std::map<std::string, float> DISJacquetBlondel(TClonesArray* tracks,
60  TClonesArray* electrons,
61  TClonesArray* photons,
62  TClonesArray* neutral_hadrons)
63 {
64  // Jacquet-Blondel method:
65  float delta_track = 0.0;
66  auto temp_p = TVector3();
67  for (int i = 0 ; i < tracks->GetEntries(); i++) {
68  auto track_mom = static_cast<Track*>(tracks->At(i))->P4();
69  if (isnan(track_mom.E()))
70  continue;
71  delta_track += (track_mom.E() - track_mom.Pz());
72  temp_p = temp_p + track_mom.Vect();
73  }
74

76  if (electrons->GetEntries()>0) {

78  if (!isnan(e.E())) {
79  delta_track_noel = delta_track - (e.E() - e.Pz());

81  }
82

86  if (isnan(pf_mom.E()))
87  continue;
88  delta_photon += (pf_mom.E() - pf_mom.Pz());
89  temp_p = temp_p + pf_mom.Vect();
90  }
91 
92  float delta_neutral = 0;
93  float delta_neutral_noBarrel = 0;
94  auto ptmiss_noBarrel = TVector3();
95 
96  for (int i = 0; i < neutral_hadrons->GetEntries(); i++) {
97  auto pf_mom = static_cast<Tower*>(neutral_hadrons->At(i))->P4();
98  if (isnan(pf_mom.E()))
99  continue;
100  delta_neutral += (pf_mom.E() - pf_mom.Pz());
101  temp_p = temp_p+ pf_mom.Vect();
102  if (TMath::Abs(pf_mom.Eta())>1.0) {
103  delta_neutral_noBarrel += (pf_mom.E() - pf_mom.Pz());
104  ptmiss_noBarrel = ptmiss_noBarrel + pf_mom.Vect();
105  }
106  }
107

111  float delta_noBarrel = delta_track + delta_photon + delta_neutral_noBarrel;
112 
113  float y_JB = delta/(2.0*10.0);
114  auto ptmiss = temp_p.Perp();
115  float Q2_JB = (ptmiss*ptmiss)/(1.0-y_JB);
116  float s = 4.0*10.0*275.0;
117  float x_JB = Q2_JB/(s*y_JB);

120  dis_variables["x_JB"] = x_JB;

131 inline float sIP3D(Jet* jet, Track* track)
132 {
133  const TLorentzVector &jetMomentum = jet->P4();
134  float jpx = jetMomentum.Px();
135  float jpy = jetMomentum.Py();
136  float jpz = jetMomentum.Pz();
137 
138  float d0 = TMath::Abs(track->D0);
139 
140  float xd = track->Xd;
141  float yd = track->Yd;
142  float zd = track->Zd;
143  float dd0 = TMath::Abs(track->ErrorD0);
144  float dz = TMath::Abs(track->DZ);
145  float ddz = TMath::Abs(track->ErrorDZ);
146 
147  int sign = (jpx * xd + jpy * yd + jpz * zd > 0.0) ? 1 : -1;
148  //add transverse and longitudinal significances in quadrature
149  float sip = sign * TMath::Sqrt(TMath::Power(d0 / dd0, 2) + TMath::Power(dz / ddz, 2));
150 
151  return sip;
152 }
