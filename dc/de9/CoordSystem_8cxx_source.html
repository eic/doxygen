63 int CoordSystem::calculateExpansionRules(int **expansion)
64 {
65  int idx = 0;
66 
67  // If all 3 coordinates are used, no expansion needed; but reset pointer to NULL;
68  if (coord_num == 3)
69  {
70  *expansion = 0;
71  return 0;
72  } /*if*/
73 
74  int *ptr = *expansion = (int*)malloc(3*sizeof(int));
75  if (!ptr) return -1;
76 
77  for(int ip=0; ip<3; ip++)
78  if (fake[ip])
79  ptr[ip] = -1;
80  else
81  ptr[ip] = idx++;
82 
83  return 0;
84 } /* CoordSystem::calculateExpansionRules */

92 void CoordSystem::projectToLocalCoordinates(double in[], double out[])
93 {
94  int idx= 0;
95  double buffer[3], *ptr;
96 
97  // Save all 3 components if in[] and out[] is the same pointer;
